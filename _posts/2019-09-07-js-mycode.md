## instanceof

```js
// 遍历__proto__ 属性和 parent.prototype比较
function instanceof(child, parent){
  while(1){
    if(child === null){
      return false
    }
    if(child.__proto__ === parent.prototype){
      return true
    }
    child = child.__proto__
  }
}
```

```js
function create(proto){
```

## debounce 防抖

```js
function debounce(func, time){
  var timer = null
  return function(...args) {
    clearTimeout(timer)
    timer = setTimeout(()=>{
      func(...args)
    }, time)
  }
}
```

## 节流

```js
function lessTime(func, time){
  var bool = false
  return (...args) => {
    if(bool) return
    setTimeout(()=>{
      bool = true
      func.call(this, ...args)
    }, time)
  }
}
```

## call

- 将函数设置成对象的属性
- 调用
- 将结果返回

```js
Function.prototype.callA = function(obj, ...args){
  obj = Object(obj) || window
  obj.fn = this
  var result = obj.fn(...args)
  Reflect.deleteProperty(obj, 'fn')
  return result
}
```

## new

- 创建一个全新的对象
- 执行原型的构造方法
- this指向创建的对象
- 原型链继承
- 如果是基本类型，copy个副本

```js
function newObj(parent, ...args){
  var obj = new Object()
  obj.__proto__ = parent.prototype
  // 绑定this
  var ret = parent.call(obj, ...args)
  return typeof ret === 'object' ? ret : obj
}
```
