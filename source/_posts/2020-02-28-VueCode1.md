# 简写Vue代码 -- 数据响应式

> 本章记录简版Vue 数据响应式代码，注释会标注作用。如缺少重要知识点，👉🏻[欢迎交流](https://github.com/qq240814476/ml-blog/issues)

```js
// 简版 Vue 数据监听代码
class MVue{
  constructor(options){
    this.$options = options
    this.$data = options.data
    // observe 调用大名鼎鼎的 defineReative 重写 get set 监听数据
    this.observe(this.$data)

    // 数据监听完就开始编译了
    new Compile(options.el, this)

    // 都干完了调用生命周期钩子
    if(options.create){
      options.create.call(this)
    }
  }

  observe(data){
    if(!data || typeof data !== 'object') return
    Object.keys(data).forEach(key => {
      this.defineReactive(data, key, data[key])

      // 使用 this.$data.XXX 不够优雅，代理下 使用this.XXX 请求
      this.proxyData(key)
    })
  }

  defineReactive(data,key, value){
    // 递归调用，遍历所有属性
    this.observe(value)

    //! 每个属性 1:1 对应一个 Dep
    var dep = new Dep()
    Object.defineProperty(data, key, {
      get() {
        //! 依赖收集 Dep.target 实际上就是watcher, 见 Watcher 类
        Dep.target && dep.addDep(Dep.target)
        return value
      },
      set(val) {
        console.log('修改了',key,'设置成', val)
        value = val

        //! 通知watcher更新
        dep.notify()
      }
    })
  }

  // this.$data.XXX => this.XXX
  proxyData(key){
    Object.defineProperty(this,key, {
      get(){
        return this.$data[key]
      },
      set(val){
        if(val === this.$data[key]) return
        this.$data[key] = val
      }
    })
  }
}

// 发布 -- 订阅 核心类
// 收集一波 watcher 相当于订阅 数据变动时 Dep 通知订阅的watcher 相当于发布
class Dep{
  constructor(){
    this.deps = []
  }

  addDep(watcher){
    this.deps.push(watcher)
  }

  notify(){
    this.deps.forEach(watcher => {
      watcher.update()
    })
  }
}

// 与vm绑定实现update，实际原理没这么简单，这只是简化版
// 真实的是直接重调 整个 update 然后patch 利用diff算法 替换改变的vdom
class Watcher{
  constructor(vm, key, updater){
    this.vm = vm
    this.key = key
    this.updater = updater

    Dep.target = this
    this.vm[key]
    Dep.target = null
  }

  update(){
    this.updater.call(this.vm, this.vm[this.key])
  }
}
```
